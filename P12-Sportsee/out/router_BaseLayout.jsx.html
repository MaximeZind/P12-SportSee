import React from 'react';

 * @returns {JSX.Element} La représentation JSX du composant BaseLayout.
 */


function BaseLayout() {

    return (
        <>
            <Nav />
            <main>
                <SideMenu />
                < Outlet />
            </main>
        </>
    );
}


export default BaseLayout;
